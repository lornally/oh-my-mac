## 阶段二建议配置

```json
{
  "files.exclude": {
    // Python 缓存
    "**/__pycache__": true,
    "**/.mypy_cache": true,
    
    // 虚拟环境
    "**/.venv": true,
    "**/.venv_tools": true,
    "**/venv": true,
    
    // Node.js
    "**/node_modules": true,
    
    // 构建输出
    "**/build": true,
    "**/dist": true,
    "**/public/build": true,
    
    // 废弃项目
    "**/废弃/**": true,
    "**/*废弃*/**": true
  },
  "files.watcherExclude": {
    "**/__pycache__/**": true,
    "**/.mypy_cache/**": true,
    "**/.venv/**": true,
    "**/.venv_tools/**": true,
    "**/node_modules/**": true,
    "**/build/**": true,
    "**/dist/**": true,
    
    // 脑图项目的 log 目录
    "**/脑图_keydog_贱狗_已经独立/log/**": true,
    "**/脑图_keydog_贱狗_已经独立/参考/**/build/**": true,
    "**/脑图_keydog_贱狗_已经独立/参考/**/__pycache__/**": true
  }
}
```
